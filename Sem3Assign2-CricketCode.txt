void Sort(struct Player_Info p[5] , int n,int function){
    float Average[n];
    for(int h=0; h<n ; h++){
        Average[h]=p[h].Avg;
    }
    int i ,j;
    float key;
    for (i = 1; i < n; i++) {
        key = Average[i];
        j = i - 1;

        while (j >= 0 && Average[j] > key) {
          Average[j+1] = Average[j];
            j = j - 1;
        }
        Average[j+1] = key;
    }

if(function==1){
int count=0;
for(int i=n-1 ; i>=0 ; i--){
    for(int j=0 ; j<n ; j++){
        if(Average[i]==p[j].Avg){
            count++;
            printf("\n %d) %s",count,p[j].Name);
        }
    }
}
}

else if(function==2){
    for(int i=0; i<n ; i++){
        if(Average[n-1]==p[i].Avg){
            printf("%s" , p[i].Name);
        }
    }
}
